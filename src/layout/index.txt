<script setup lang="ts">
import { computed, watch } from 'vue'
import Sidebar from '@/layouts/components/Sidebar/Sidebar.vue'
import TopHeader from '@/layouts/components/TopHeader/Topheader.vue'
import { useAppStore } from '@/store/modules/app'

const appStore = useAppStore()

// 计算侧边栏宽度
const sidebarWidth = computed(() => {
  return appStore.isCollapsed ? '72px' : '240px'
})

// 添加对移动端侧边栏状态的监听
watch(() => appStore.sidebarMobileOpen, (open) => {
  if (open && appStore.isMobile) {
    document.body.style.overflow = 'hidden'
  }
  else {
    document.body.style.overflow = ''
  }
})
</script>

<template>
  <div class="grid-layout-container">
    <!-- 侧边栏区域 -->
    <aside class="sidebar-area" :style="{ width: appStore.isMobile ? '240px' : sidebarWidth }">
      <Sidebar />
    </aside>

    <!-- 主内容区域 -->
    <main class="main-area">
      <!-- 顶部导航 -->
      <header class="header-area">
        <TopHeader />
      </header>

      <!-- 内容区域 -->
      <section class="content-area">
        <RouterView />
      </section>
    </main>
  </div>
</template>

<style lang="scss" scoped>
.grid-layout-container {
  display: grid;
  grid-template-columns: auto 1fr;
  grid-template-rows: 60px 1fr;
  grid-template-areas:
    "sidebar header"
    "sidebar content";
  height: 100vh;
  overflow: hidden;
  transition: grid-template-columns 0.3s ease;
  gap: 18px; /* 移除间隙 */
  background-color: #a5a2a22d;

  .sidebar-area {
    grid-area: sidebar;
    transition: width 0.3s ease;
    height: 100%;
    overflow: hidden;
    z-index: 1000;
  }

  .header-area {
    grid-area: header;
    height: 60px;
    z-index: 100;
  }

  .content-area {
    grid-area: content;
    overflow: auto;
    padding: 20px;
    background-color: var(--el-bg-color-page);
  }
}

// 响应式设计 - 移动端适配
@media (max-width: 768px) {
  .grid-layout-container {
    grid-template-columns: 1fr;
    grid-template-areas:
      "header"
      "content";

    .sidebar-area {
      position: fixed;
      left: -240px;
      top: 0;
      bottom: 0;
      transition: left 0.3s ease;
      z-index: 1000;

      &.mobile-open {
        left: 0;
      }
    }
  }
}
</style>
